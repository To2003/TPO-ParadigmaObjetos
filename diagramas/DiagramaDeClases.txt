    class Partida {
        -estado: EstadoPartida
        -turnoActual: int
        +iniciarJuego()
        +procesarAccion(accion: String)
        +guardarPartida()
        +cargarPartida()
    }

    class Mapa {
        -nodos: List~NodoMapa~
        -nodoActual: NodoMapa
        +generarMapa(seed: long)
        +getNodoActual() NodoMapa
        +avanzar(nodo: NodoMapa)
    }

    class NodoMapa {
        -tipo: TipoNodo
        -conexiones: List~NodoMapa~
        -visitado: boolean
        +getTipo() TipoNodo
        +getConexiones() List
    }

    class TipoNodo {
        <<enumeration>>
        COMBATE
        ELITE
        JEFE
        TIENDA
        DESCANSO
        TESORO
    }

    class Party {
        -personajes: List~Personaje~
        -inventarioComun: Inventario
        +agregarPersonaje(p: Personaje)
        +estaViva() boolean
        +getPersonajesVivos() List
    }

    class Personaje {
        <<abstract>>
        -nombre: String
        -hp: int
        -hpMax: int
        -pa: int
        -paMax: int
        -ataque: int
        -defensa: int
        -velocidad: int
        -nivel: int
        -exp: int
        +calcularDanoBase() int
        +recibirDano(dmg: int)
        +consumirPA(costo: int)
        +activarDefensa()
        +subirNivel()
        +ganarExp(cantidad: int)
    }

    class Rogue {

    class Habilidad {
        -nombre: String
        -descripcion: String

        -tipoObjetivo: TipoObjetivo
        -efecto: EfectoEstado
        +usar(usuario: Personaje, objetivo)
        +getCostoPa() int
    }

    class Inventario {
        -items: List~Item~
        -capacidadMax: int
        +agregarItem(item: Item)
        +removerItem(item: Item)
        +usarItem(item: Item, objetivo)
    }

    class Item {
        <<abstract>>
